    ];


    function renderTable() {

        const tbody = document.getElementById('inventoryBody');

        tbody.innerHTML = "";

        

        let alertCount = 0;

        

        inventory.forEach((item, index) => {

            let isLow = item.stock <= item.minLimit;

            if (isLow) alertCount++;

            

            let statusBadge = isLow 

                ? `<span class="badge lowstock"><i class="fa-solid fa-triangle-exclamation"></i> Low Stock</span>`

                : `<span class="badge instock"><i class="fa-solid fa-circle-check"></i> In Stock</span>`;


            tbody.innerHTML += `

                <tr>

                    <td style="color: var(--text-silver); font-family: monospace; font-size: 0.95rem;">${item.barcode}</td>

                    <td style="font-weight: 600; font-size: 1.05rem;">${item.name}</td>

                    <td class="price-text">${item.price.toFixed(2)} JOD</td>

                    <td style="font-size: 1.1rem; font-weight: 700; color: ${isLow ? 'var(--danger-red)' : '#fff'}">${item.stock} units</td>

                    <td>${statusBadge}</td>

                    <td>

                        <button class="action-btn" onclick="updateStock(${index}, 1)"><i class="fa-solid fa-plus"></i> Restock</button>

                        <button class="action-btn sell-btn" onclick="updateStock(${index}, -1)"><i class="fa-solid fa-cart-shopping"></i> Sell Unit</button>

                    </td>

                </tr>

            `;

        });

        

        document.getElementById('totalItems').innerText = inventory.length;

        document.getElementById('alertItems').innerText = alertCount;

    }


    function updateStock(index, amount) {

        if (inventory[index].stock + amount >= 0) {

            inventory[index].stock += amount;

            renderTable();

        }

    }


    function searchMedicine() {

        let input = document.getElementById("searchInput").value.toLowerCase();

        let rows = document.getElementById("inventoryBody").getElementsByTagName("tr");

        

        for (let i = 0; i < rows.length; i++) {

            let nameColumn = rows[i].getElementsByTagName("td")[1];

            let barcodeColumn = rows[i].getElementsByTagName("td")[0];

            if (nameColumn || barcodeColumn) {

                let nameText = nameColumn.textContent || nameColumn.innerText;

                let barcodeText = barcodeColumn.textContent || barcodeColumn.innerText;

                if (nameText.toLowerCase().indexOf(input) > -1 || barcodeText.toLowerCase().indexOf(input) > -1) {

                    rows[i].style.display = "";

                } else {

                    rows[i].style.display = "none";

                }

            }

        }

    }
